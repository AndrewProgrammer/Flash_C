#include <time.h>
#include <iostream>
using namespace std;
int a,b,c;
double sum()
{
	cout << "Введите 2 числа:\n";
	cin >> a >> b;
	c = a + b;
	cout << "Ответ" << c << endl;
	return 0;
}
double sub()
{
	cout << "Введите 2 числа:\n";
	cin >> a >> b;
	c = a - b;
	cout << "Ответ" << c << endl;
	return 0;
}
double mul()
{
	cout << "Введите 2 числа:\n";
	cin >> a >> b;
	c = a * b;
	cout << "Ответ" << c << endl;
	return 0;
}
double del()
{
	cout << "Введите 2 числа:\n";
	cin >> a >> b;
	c = a / b;
	cout << "Ответ" << c << endl;
	return 0;
}
double logic()
{
	int d;
	cout << "Введите 0,1,2,3 на ваш выбор:\n";
	cin >> d;
	if (d == 0)
	{
		sum();
	}
	else if (d == 1)
	{
		sub();
	}
	else if (d == 2)
	{
		mul();
	}
	else if (d == 3)
	{
		del();
	}
	else cout << "Error!" << endl;
	return 0;
}
void main()
{
	setlocale(0, "ru");
	logic();
}
