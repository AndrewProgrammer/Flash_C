#include <iostream>
#include <ctime>
using namespace std;
const int size = 10;
double averege()
{
	int a[size], min = -10, max = 10;
	srand(time(0));
	rand();
	for (int i = 0; i < size; i++)
	{
		a[i] = rand() % (max - min + 1) + min;
	}
	cout << "Массив из 10 элементов:\n";
	for (int i = 0; i < size; i++)
	{
		cout << a[i] << " ";
	}
	double poz = 0, neg = 0, null = 0;
	for (int i = 0; i < size; i++)
	{
		if (a[i] > 0)
			poz++;
		else if (a[i] < 0)
			neg++;
		else
			null++;
	}
	cout << endl;
	cout << "Количевство положительных элементов массива " << poz << endl;
	cout << "Количевство отрицательных элементов массива " << neg << endl;
	cout << "Количевство нулей в массиве " << null << endl;
	return 0;
}

void main()
{
	setlocale(0, "ru");
	averege();
}
