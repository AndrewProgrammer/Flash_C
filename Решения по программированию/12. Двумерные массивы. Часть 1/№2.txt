#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	srand(time(0));
	int b[5][5], min = -5, max = 5, pol = 0, nepol = 0, null = 0;
	for (int y = 0; y < 5; y++)
	{
		for (int x = 0; x < 5; x++)
		{
			b[y][x] = rand() % (max - min + 1) + min;
		}
	}
	for (int y = 0; y < 5; y++)
	{
		for (int x = 0; x < 5; x++)
		{
			cout.width(5);
			cout << b[y][x];
		}
		cout << endl;
	}
	for (int y = 0; y < 5; y++)
	{
		for (int x = 0; x < 5; x++)
		{
			if (b[y][x] > 0)
			{
				pol++;
			}
			else if (b[y][x] < 0)
			{
				nepol++;
			}
			else
			{
				null++;
			}
		}
	}
	cout << "Количевство положительных элементов в двумерном массиве =  " << pol << endl;
	cout << "Количевство отрицательных элементов в двумерном массиве  = " << nepol << endl;
	cout << "Количевство нулевых элементов в двумерном массиве = " << null << endl;
}
