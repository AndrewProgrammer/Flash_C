#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	srand(time(0));
	int b[5][5], min = -100, max = 100;
	int i,j, index_min = 0, index_max = 0, index_min1 = 0, index_max1 = 0,sum = 0;
	for (int y = 0; y < 5; y++)
	{
		for (int x = 0; x < 5; x++)
		{
			b[y][x] = rand() % (max - min + 1) + min;
		}
	}
	for (int y = 0; y < 5; y++)
	{
		for (int x = 0; x < 5; x++)
		{
			cout.width(5);
			cout << b[y][x];
		}
		cout << endl;
	}
	int min1 = b[0][0],max1 = b[0][0];
	for (int i = 0; i < 5; i++)
	{
		for (int j = 0; j < 5; j++)
		{
			cout << "Это элемент массива[" << i << "] [" << j << "]: " << b[i][j] << "\n";
		}
	}
	cout << endl;
	for (i = 1; i < 5; i++)
	{
		for ( j = 1; j < 5; j++)
		{
			if (b[i][j] < min1)
			{
				min1 = b[i][j];
				index_min = i;
				index_min1 = j;
			}
			else if (b[i][j] > max1)
			{
				max1 = b[i][j];
				index_max = i;
				index_max1 = j;
			}
		}
	}
	cout << "Номер элемента массива с минимальным значением : " << index_min << " " << index_min1 << "(" << min1 << ")" << endl;
	cout << "Номер элемента массива с максимальным значением: " << index_max << " " << index_max1 << "(" << max1 << ")" << endl;
	cout << endl;
	if (index_min<index_max && index_min1 < index_max1)
		for (i = index_min + 1; i < index_max; i++)
		{
			for (j = index_min1 + 1; j < index_max; j++)
			{
				sum += b[i][j];
			}
		}
	else
		for (i = index_max + 1; i < index_min; i++)
		{
			for (j = index_max1 + 1; j < index_min1; j++)
			{
				sum += b[i][j];
			}
		}
	cout << "Сумма значений элементов,находящихся между элементами с минимальным и максимальным значением: " << sum << "\n";
}
