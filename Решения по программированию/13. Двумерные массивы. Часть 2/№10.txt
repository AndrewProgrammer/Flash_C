#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	double M[3][3];
	double M2[3][3];
	double M3[3][3];
	cout << "Заполните 1 матрицу 3х3:" << endl;
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			cin >> M[i][j];
			M3[i][j] = 0;
		}
	}
	cout << "Заполните 2 матрицу 3х3:" << endl;
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			cin >> M2[i][j];
		}
	}
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			for (int m = 0; m < 3; m++)
			{
				M3[i][j] += M[i][m] * M2[m][j];
			}
		}
	}
	for (int i = 0; i < 3; i++)
	{
		cout << "| ";
		for (int j = 0; j < 3; j++)
		{
			cout << M[i][j] << " ";
		}
		cout << "|      |";
		for (int k = 0; k < 3; k++)
		{
			cout << M2[i][k] << " ";
		} 
		cout << "|";
		cout << endl;
	}
	cout << endl;
	cout << "Результативная матрица:" << endl;
	for (int i = 0; i < 3; i++)
	{
		cout << "| ";
		for (int j = 0; j < 3; j++)
		{
			cout << M3[i][j] << " ";
		}
		cout << "|        " << endl;
	}
}
