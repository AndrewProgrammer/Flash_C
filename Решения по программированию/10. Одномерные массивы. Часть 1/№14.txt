#include <iostream>
#include <time.h>
using namespace std;
void main()
{
	setlocale(0, "ru");
	const int n = 20,N = 10;
	int a[n];
	srand(time(0));
	rand();
	for (int k = 0; k < n; k++)
	{
		a[k] = rand() % 201;
	}
	cout << "Массив с рандомными числами :\n";
	for (int j = 0; j < n; j++)
	{
		cout << a[j] << " ";
	}
	cout << endl;
	int	p = 0, p1 = 0, p2 = 0;
	for (int i = 0; i < n; i++)
	{
		if (a[i] >= 0 && a[i] < 10)
		{
			p+=1;
		}
		else if (a[i] >= 10 && a[i] < 100)
		{
			p1+=1;
		}
		else if (a[i] >= 100 && a[i] <= 200)
		{
			p2+=1;
		}
	}
	cout << "Количевство цифр 1 разрядности: " << p << endl;
	cout << "Количевство цифр 2 разрядности: " << p1 << endl;
	cout << "Количевство цифр 3 разрядности: " << p2 << endl;
}
