#include<iostream>
#include <time.h>
using namespace std;
void main()
{
	setlocale(0, "ru");
	srand(time(0));
	rand();
	const int n = 10;
	int a[n],b[n];
	for (int k = 0; k < n; k++)
	{
		a[k] = rand() % 101;
	}
	cout << "Массив с рандомными числами:\n";
	for (int j = 0; j < n; j++)
	{
		cout << a[j] << ' ';
	}
	cout << endl;
	for (int i = 0, j = 9; i < 10; i++, j--)
	{
		b[i] = a[j];
	}
	cout << "Массив с изменненым порядком чисел:\n";
	for (int i = 0; i < 10; i++)
	{
		cout << b[i] << " ";
	}
	cout << endl;
}
