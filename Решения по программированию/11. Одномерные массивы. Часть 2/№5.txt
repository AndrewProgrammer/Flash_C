#include <iostream>
#include <time.h>
using namespace std;
void main()
{
	setlocale(0, "ru");
	const int size = 10;
	int a[size], min = 0, max = 50, pos = -1;
	double  sum = 0, avg;
	srand(time(0));
	rand();
	for (int k = 0; k < size; k++)
	{
		a[k] = rand() % (max - min + 1) + min;
	}
	cout << "Массив с рандомными числами: \n";
	for (int j = 0; j < size; j++)
	{
		cout << a[j] << ' ';
	}
	cout << endl;
	for (int i = 0; i < size; i++)
	{
		sum += a[i];
	}
	avg = sum / size;
	int Min = 0;
	for (int i = 1; i < size; i++)
	{
		if (abs(a[i] - avg)<abs(a[Min] - avg))
		{
			Min = i;
		}
	}
	cout << "Avg: " << avg << endl;
	cout << "Число которое приблизительно равно Avg: " << a[Min] << endl;
}
