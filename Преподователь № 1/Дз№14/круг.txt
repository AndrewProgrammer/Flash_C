#include <iostream>
using namespace std;
int main()
{
	setlocale(0, "ru");
	double r, y1, y2;
	cout << "¬ведите радиус: ";
	cin >> r;
	cout << r << endl;
	for (int i = 1; i <= 2 * r+5; i++)
	{
		for (int j = 1; j <= 2 * r+5; j++)
		{
			y1 = r + sqrt(r*r - pow(r - j, 2));
			y2 = r - sqrt(r*r - pow(r - j, 2));
			if (i >= y2 && i <= y1)
				cout << "*";
			else cout << " ";
		}
		cout << endl;
	}
}
