#include <iostream>
using namespace std;
int main()
{
	setlocale(0, "ru");
	const int size = 5;
	int A[size] = { 1, 2, 3, 4, 5 }, B[size] = { 10, 11, 12, 13, 14 }, C[10];
	cout << "1 Массив с числами: \n";
	for (int i = 0; i < size; i++)
	{
		cout << A[i] << ' ';
	}
	cout << endl;
	cout << "2 Массив с числами: \n";
	for (int o = 0; o < size; o++)
	{
		cout << B[o] << ' ';
	}
	cout << endl;
	cout << "3 массив: \n";
	for (int j = 0; j < 10; j+=2)
	{
		C[j] = A[j / 2];
		C[j + 1] = B[j / 2];
	}
	for (int p = 0; p < size * 2; p++)
	{
		cout << C[p] << ' ';
	}
}
