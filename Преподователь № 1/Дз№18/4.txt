#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	const int size = 5;
	int A[size], B[size], C[size * 2] = { 1, 10, 2, 11, 3, 12, 4, 13, 5, 14 };
	cout << "3 Массив с числами: \n";
	for (int o = 0; o < size * 2; o++)
	{
		cout << C[o] << ' ';
	}
	cout << endl;
	cout << "1 и 2 массив: \n";
	for (int j = 0; j < size; j++)
	{
		A[j] = C[j * 2];
		B[j] = C[j * 2 + 1];
	}
	for (int p = 0; p < size; p++)
	{
		cout << A[p] << ' ' << B[p] << ' ';
	}
}
