#include <iostream> 
using namespace std;
const int N = 10; // размер массива
double mas()
{
	int i, min_val, max_val;
	int index_min = 0;
	int index_max = 0;
	int sum = 0;
	int nums[N], min = -10, max = 10;
	for (i = 0; i < N; i++)
		nums[i] = rand() % (max - min + 1) + min;
	for (i = 0; i < N; i++)
		cout << "Это элемент массива[" << i << "]: " << nums[i] << endl;
	min_val = max_val = nums[0];
	for (i = 1; i<N; i++)
	{
		if (nums[i]<min_val)
		{
			min_val = nums[i];
			index_min = i;
		}
		if (nums[i]>max_val)
		{
			max_val = nums[i];
			index_max = i;
		}
	}
	cout << endl;
	cout << "Номер элемента массива с минимальным значением : " << index_min << "(" << min_val << ")" << endl;
	cout << "Номер элемента массива с максимальным значением: " << index_max << "(" << max_val << ")" << endl;
	return 0;
}
void main()
{
	setlocale(0, "ru");
	mas();
}
