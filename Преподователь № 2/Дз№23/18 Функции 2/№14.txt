#include <iostream>
using namespace std;
char toUpper(char c)
{
	if (c >= 97 && c <= 122)
	{
		c -= 32;
		return c;
	}
	else
	{
		return 0;
	}
}
char toLower(char ch)
{
	if (ch >= 65 && ch <= 90)
	{
		ch += 32;
		return ch;
	}
	else
	{
		return 0;
	}
}
void main()
{
	setlocale(0, "ru");
	char a;
	cout << "Enter digit\n";
	cin >> a;
	if (a >= 'a' && a <= 'z')
	{
		cout << "Символ в верхнем регистре: " << toUpper(a) << endl;
	}
	else if (a >= 'A' && a <= 'Z')
	{
		cout << "Символ в нижнем регистре: " << toLower(a) << endl;
	}
	else 
	{
		cout << "Error!\n";
		exit(0);
	}
}
