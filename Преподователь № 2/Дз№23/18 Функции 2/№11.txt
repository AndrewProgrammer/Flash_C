#include <ctime>
#include <iostream>
using namespace std;
int BinarSearch(int x[], int k, int key)
{
	int start = 0, end = k - 1;
	int middle = (start + end) / 2;
	while (start <= end)
	{
		if (key == x[middle])
		{
			return key;
		}
		else if (key > x[middle])
		{
			start = middle + 1;
			return key;
		}
		else if (key < x[middle])
		{
			end = middle - 1;
			return key;
		}
		else
		{
			return -1;
			exit(0);
		}
	}
	return key;
}
void main()
{
	setlocale(0, "ru");
	srand(time(0));
	const int size = 10;
	int b[size], min = -10, max = 10, key;
	for (int i = 0; i < size; i++)
	{
		b[i] = rand() % (max - min + 1) + min;

	}
	cout << "Массив с рандомными числами: \n";
	for (int i = 0; i < size; i++)
	{
		cout << b[i] << ' ';
	}
	cout << endl;
	cout << "Enter key: \n";
	cin >> key;
	cout << "Key found = " <<  BinarSearch(b, size, key) << endl;
}
