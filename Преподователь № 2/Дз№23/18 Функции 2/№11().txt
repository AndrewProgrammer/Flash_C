#include <ctime>
#include <iostream>
using namespace std;
int BinarySearch(int x[], int k, int key){
	bool found = false;
	int high = k - 1, low = 0;
	int middle = (high + low) / 2;
	while (!found && high >= low){
		if (key == x[middle])
			found = true;
		else if (key < x[middle])
			high = middle - 1;
		else
			low = middle + 1;
		middle = (high + low) / 2;
	}
	return found ? middle : -1;
}
void main()
{
	setlocale(0, "ru");
	srand(time(0));
	const int size = 10;
	int b[size], min = -10, max = 10, key;
	for (int i = 0; i < size; i++)
	{
		b[i] = rand() % (max - min + 1) + min;

	}
	cout << "Массив с рандомными числами: \n";
	for (int i = 0; i < size; i++)
	{
		cout << b[i] << ' ';
	}
	cout << endl;
	cout << "Enter key: \n";
	cin >> key;
	cout << "Key found = " <<  BinarySearch(b, size, key) << endl;
}
