#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	srand(time(0));
	rand();
	const int size = 10;
	int b[size], min, max;
	cout << "Введите диапозон массива(2 значения): \n";
	cin >> min >> max;
	for (int k = 0; k < size; k++)
	{
		b[k] = rand() % (max - min + 1) + min;
	}
	cout << "Массив с рандомными числами:\n";
	for (int j = 0; j < size; j++)
	{
		cout << b[j] << ' ';
	}
	cout << endl;
	for (int i = 1; i < size; i++)
	{
		for (int j = i; j > 0 && b[j] < b[j - 1]; j--)
		{
			swap(b[j], b[j - 1]);
		}
	}
	cout << "Отсортированый массив методом вставками:\n";
	for (int k = 0; k < size; k++)
	{
		cout << b[k] << ' ';
	}
	cout << endl;
}
