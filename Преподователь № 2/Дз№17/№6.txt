#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	const int size = 10;
	int b[size], min, max;
	cout << "Введите диаопозон значений: \n";
	cin >> min >> max;
	srand(time(0));
	rand();
	for (int k = 0; k < size; k++)
	{
		b[k] = rand() % (max - min + 1) + min;
	}
	cout << "Массив с рандомными числами: \n";
	for (int j = 0; j < size; j++)
	{
		cout << b[j] << ' ';
	}
	cout << endl;
	for (int i = 1; i < size; i++)
	{
		for (int j = i; j > 0 && b[j] < b[j - 1]; j--)
		{
			swap(b[j], b[j - 1]);
		}
	}
	cout << "Отсортированый массив методом вставками:\n";
	for (int i = 0; i < size; i++)
	{
		cout << "Это элемент массива[" << i << "]: " << b[i] << "\n";
	}
	cout << endl;
	int a;
	cout << "Введите элемент массива : \n";
	cin >> a;
	int start = 0, end = size - 1, pos = -1, start1 = 0, end1 = size - 1, pos1 = -1;;
	while (start <= end)
	{
		int middle = (start + end) / 2;
		if (a == b[middle])
		{
			pos = middle;
			break;
		}
		else if (a > b[middle])
		{
			start = middle + 1;
		}
		else
		{
			end = middle - 1;
		}
	}
	cout << "Его индекс: \n" << pos << endl;
	int c;
	cout << "Введите элемент массива которым хотите заменить : \n";
	cin >> c;
	while (start1 <= end1)
	{
		int middle1 = (start1 + end1) / 2;
		if (c == b[middle1])
		{
			pos1 = middle1;
			break;
		}
		else if (c > b[middle1])
		{
			start1 = middle1 + 1;
		}
		else
		{
			end1 = middle1 - 1;
		}
	}
	cout << "Его индекс: \n" << pos1 << endl;
	swap(b[pos1], b[pos]);
	for (int k = 0; k < size; k++)
	{
		cout << "Это элемент массива[" << k << "]: " << b[k] << "\n";
	}
}
