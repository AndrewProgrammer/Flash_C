#include <time.h>
#include <iostream>
using namespace std;
void PrintArray(int *mas, int size)
{
	for (int i = 0; i < size; i++, mas++)
	{
		cout << *mas << endl;
	}
}
void FieldArray(int *mas, int size)
{
	for (int i = 0; i < size; i++, mas++)
	{
		*mas = rand() % 10 + 1;
	}
}
int Sum1(int *mas, int size)
{
	int sum = 0;
	for (int i = 0; i < size; i++,mas++)
	{
		if (*mas > 0)
		{
			sum += *mas;
		}
	}
	return sum;
	
}
void main()
{
	setlocale(0, "ru");
	srand(time(0));
	int mas[10];
	FieldArray(mas, 10);
	PrintArray(mas, 10);
	cout << "—умма элементов массива которые €вл€ютс€ положительными = " << Sum1(mas, 10) << endl;
	system("pause");
}
