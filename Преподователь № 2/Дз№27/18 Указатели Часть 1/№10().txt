using namespace std;
void PrintArray(int *mas, int size)
{
	for (int i = 0; i < size; i++, mas++)
	{
		cout << *mas << endl;
	}
}
void FieldArray(int *mas, int size)
{
	for (int i = 0; i < size; i++, mas++)
	{
		*mas = rand() % 10 + 1;
	}
}
void Shift(int *mas, int size)
{
	int shift;
	cout << "¬ведите число сдвига\n";
	cin >> shift;
	int b;
	cout << "≈сли вы ведете 1 то сдвиг массива произойдет в право.≈сли вы не наберете 1 то в противном случае сдвиг произойдет в лево: \n";
	cin >> b;
	if (b == 1)
	{
		for (int k = 0; k < shift; k++, mas++)
		{
			int temp = *(mas + size - 1);
			for (int p = size - 1; p > 1; p--, mas--)
			{
				*mas = *mas - 1;
			}
			*mas = temp;
		}
	}
	else
	{
		for (int k = 0; k < shift-1; k++, mas++)
		{
			int temp = *mas;
			for (int p = 0; p < size - 1; p++, mas++)
				*mas = *mas + 1;
			*mas= temp;
		}

	}
}
