#include <iostream>
#include <ctime>
using namespace std;
void main()
{
	srand(time(0));
	setlocale(0, "ru");
	int b[5][5], min = 0, max = 10;
	for (int y = 0; y < 5; y++)
	{
		for (int x = 0; x < 5; x++)
		{
			b[y][x] = rand() % (max - min + 1) + min;
		}
	}
	for (int k = 0; k < 5; k++)
	{
		int sum = 0;
		for (int j = 0; j < 5; j++)
		{
			cout.width(5);
			cout << b[j][k];
			if (b[k][j] >= 0)
			{
				sum += b[k][j];
			}
		}
		cout << " --- " << sum << " = ףללא " << k << " -מדמ סעמכבצא" << endl;
	}
	cout << endl;
	int i, j, p, k;
	for (p = 0; p < 5 - 1; p++)
	{
		for (k = p + 1; k < 5; k++)
		{
			if (b[5 - 1][k] < b[5 - 1][p])
			{
				for (i = 0; i < 5; i++)
				{
					swap(b[i][p], b[i][k]);
				}

			}

		}
	}
	for (i = 0; i < 5; i++)
	{
		for (j = 0; j < 5; j++)
		{
			cout.width(5);
			cout << b[i][j];
		}
		cout << endl;
	}
}
