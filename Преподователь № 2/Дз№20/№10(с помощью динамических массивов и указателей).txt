#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	int N, M, K,D;
	do
	{
		cout << "Введите размерность 1 матрицы:" << endl;
		cin >> N >> M;
		cout << "Введите размерность 2 матрицы:" << endl;
		cin >> K >> D;
		if (M != K)
		{
			system("cls");
			cout << "Невозможно умножить!" << endl;
		}
	} while (M != K);
	int **M1 = new int *[N];
	for (int i = 0; i < N; i++)
	{
		M1[i] = new int [M];
	}
	int **M2 = new int *[K];
	for (int i = 0; i < K; i++)
	{
		M2[i] = new int[D];
	}
	int **M3 = new int *[N];
	for (int i = 0; i < N; i++)
	{
		M3[i] = new int[D];
	}
	cout << "Заполните 1 матрицу:" << N << "x" << M << endl;
	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < M; j++)
		{
			cin >> M1[i][j];
		}
	}
	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < D; j++)
		{
			M3[i][j] = 0;
		}
	}
	cout << "Заполните 2 матрицу:" << K << "x" << D << endl;
	for (int i = 0; i < K; i++)
	{
		for (int j = 0; j < D; j++)
		{
			cin >> M2[i][j];
		}
	}
	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < D; j++)
		{
			for (int m = 0; m < M; m++)
			{
				M3[i][j] += M1[i][m] * M2[m][j];
			}
		}
	}
	for (int i = 0; i < N; i++)
	{
		cout << "| ";
		for (int j = 0; j < M; j++)
		{
			cout << M1[i][j] << " ";
		}
		cout << "|      |";
		for (int k = 0; k < D; k++)
		{
			cout << M2[i][k] << " ";
		} 
		cout << "|";
		cout << endl;
	}
	cout << endl;
	cout << "Результативная матрица:" << endl;
	for (int i = 0; i < N; i++)
	{
		cout << "| ";
		for (int j = 0; j < D; j++)
		{
			cout << M3[i][j] << " ";
		}
		cout << "|        " << endl;
	}
	for (int i = 0; i < N; i++)
	{
		delete[] M1[i];
	}
	delete[] M1;
	for (int i = 0; i < K; i++)
	{
		delete[] M2[i];
	}
	delete[] M2;
	for (int i = 0; i < N; i++)
	{
		delete[] M3[i];
	}
	delete[] M3;
}
