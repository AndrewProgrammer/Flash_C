template<typename T>
class roctor
{
	template<typename IT>
	vector(IT start, IT end)	
	{
		for(;start!=end;++start)
		{
			push_back(*start);
		}		
	}
};
void main()
{
	int ar[] = {1,2,3,4,5};
	vector<int> v(ar,ar+2);//За концом диапозона, то есть будет (1,2) на выводе.
}
-------------------------------------
Методы вектора:
T& v.front();//Возвращает константную ссылку на первый элемень в векторе, если вектор пустой то не предвиденное поведение(UB) программы(то есть ошибка)
T& v.back();
T& v.assign();//Полная перезапись
T& v.pop_back();//Метод который удаляем последний элемент у вектора
T& v.push_back();//Добавления вектора.
iterator v.insert(iterator position, T val);//возвращает итератор на первый добавленный элемент
iterator v.insert(iterator position,size_t n, T val);//Вставка диапозона
iterator erase(iterator position)//Метод удаления элемента в классе вектор
iterator erase(iterator start,iterator end);//Удаляет диапозон от старта до енда, не включая енд, (возвращается итератор на первый элемент за концом массива(уже смещенный))
void clear();//Очищает все
begin();//Начальный элемент
end();//Следующий элемент после последнего
rbegin();//Итерация на последнй элемент.(Происходит инверсия( то есть переворачиваем)), плюс еще оператор ++ идет от конца в начало
rend();//Итерация на  элемент перед первым элементом.(Происходит инверсия( то есть переворачиваем)), плюс еще оператор ++ идет от конца в начало
cbegin()//Константный - c
cend();// Константный - c
crbegin();//Реверсия константы
crend();//Реверсия константы
void main()
{
	v.front() = 3;
	v.back() = 4;
}
